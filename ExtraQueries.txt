SELECT LOCATION.Name, ADDRESS.Street, ADDRESS.City, ADDRESS.State, ADDRESS.Zip FROM LOCATION
INNER JOIN ADDRESS ON ADDRESS.addr_ID = LOCATION.addr_ID;

SELECT PERSON.Phone, PERSON.Fname, PERSON.Lname FROM PERSON
INNER JOIN EMERGENCY_CONTACTS ON EMERGENCY_CONTACTS.person_ID = PERSON.person_ID;

SELECT * FROM INVOICE
WHERE CHARGE > 1000
